Restricting and Sorting Data

Limit the rows that are retrieved by a query

SELECT FIRSTNAME, LASTNAME 
  FROM PERSON
 WHERE PERSONID = 1 
   AND FIRSTNAME = 'JIM' 
   AND LASTNAME = 'JAM';

Sort the rows that are retrieved by a query

SELECT FIRSTNAME, LASTNAME 
  FROM PERSON
 WHERE GENDER = 'MALE'
 ORDER BY FIRSTNAME, LASTNAME;

Use substitution variables

SELECT FIRSTNAME, LASTNAME, GENDER
  FROM PERSON
 WHERE PERSONID = &PERSONID;

SELECT FIRSTNAME, LASTNAME, GENDER
  FROM PERSON
 WHERE PERSONID = &&PERSONID;

Use the SQL row limiting clause

SELECT FIRSTNAME, LASTNAME 
  FROM PERSON
 WHERE GENDER = 'MALE'
 ORDER BY FIRSTNAME, LASTNAME
 FETCH FIRST 5 ROWS ONLY;

SELECT FIRSTNAME, LASTNAME 
  FROM PERSON
 WHERE GENDER = 'MALE'
 ORDER BY FIRSTNAME, LASTNAME
 FETCH FIRST 5 ROWS ONLY WITH TIES;

SELECT FIRSTNAME, LASTNAME 
  FROM PERSON
 WHERE GENDER = 'MALE'
 ORDER BY FIRSTNAME, LASTNAME
 FETCH FIRST 25 PERCENT ROWS ONLY;

SELECT FIRSTNAME, LASTNAME 
  FROM PERSON
 WHERE GENDER = 'MALE'
 ORDER BY FIRSTNAME, LASTNAME
OFFSET 3 FETCH NEXT 4 ONLY;

  FROM INVOICETEST

Displaying Data from Multiple Tables

Use equijoins and nonequijoins

SELECT P.FIRSTNAME, P.LASTNAME, JD.DESIGNATION 
  FROM PERSON P JOIN JOBDETAIL JD
ON P.PERSONID = JD.PERSONID;

-- EQUIJOIN: JOINING TABLES BASED ON EQUALITY CONDITION

SELECT P.FIRSTNAME, P.LASTNAME, JD.DESIGNATION 
  FROM PERSON P JOIN JOBDETAIL JD
ON P.SALARY >= JD.BASESALARY;

-- NON EQUIJOIN: JOINING TABLES BASED ON OTHER THAN EQUALITY CONDITION

Use a self-join

SELECT P.FIRSTNAME, P.LASTNAME, PP.FIRSTNAME, PP.LASTNAME
  FROM PERSON P, PERSON PP
 WHERE P.PARENTID = PP.PERSONID;

Use outer joins

SELECT SHIP_ID, SHIP_NAME, PORT_NAME
  FROM SHIPS LEFT OUTER JOIN PORTS
    ON HOME_PORT_ID = PORT_ID
 ORDER BY SHIP_ID;
 
-- LEFT OUTER JOIN

SELECT SHIP_ID, SHIP_NAME, PORT_NAME
  FROM SHIPS LEFT JOIN PORTS
    ON HOME_PORT_ID = PORT_ID
 ORDER BY SHIP_ID;

-- LEFT OUTER JOIN (OUTER OPTIONAL)

Generate a Cartesian product of all rows from two or more tables

SELECT FIRSTNAME, LASTNAME, EMPLOYERNAME
  FROM PERSON CROSS JOIN EMPLOYEE
 ORDER BY EMPLOYERNAME;

-- CARTESIAN PRODUCT IS CONSIDERED AS AN ACCIDENTAL USE OF JOIN, AS THE OUTPUT IS GENERALLY USELESS

SELECT FIRSTNAME, LASTNAME, EMPLOYERNAME
  FROM PERSON, EMPLOYEE
 ORDER BY EMPLOYERNAME;

-- ABOVE QUERY WILL ALSO GENERATE CARTESIAN PRODUCT AS ORACLE DO NOT KNOW ABOUT THE JOIN CONDITION

Create queries using the PIVOT and UNPIVOT clause

SELECT * FROM (SELECT CUSTOMER_ID, PRODUCT, QUANTITY FROM PIVOT_TEST)
PIVOT (
SUM(QUANTITY) AS SUM_QUANTITY 
FOR (PRODUCT) 
IN ('A' AS A, 'B' AS B, 'C' AS C));

SELECT CUSTOMER_ID, A_SUM_QUANTITY, B_SUM_QUANTITY, C_SUM_QUANTITY 
FROM (SELECT CUSTOMER_ID, PRODUCT,QUANTITY FROM PIVOT_TEST)
PIVOT (
SUM(QUANTITY) AS SUM_QUANTITY 
FOR (PRODUCT) 
IN ('A' AS A, 'B' AS B, 'C' AS C));

SELECT * FROM PIVOT_TEST -- TABLE
PIVOT (
SUM(QUANTITY) AS SUM_QUANTITY 
FOR (PRODUCT) 
IN ('A' AS A, 'B' AS B, 'C' AS C)); 
-- ALTHOUGH THIS WORKS BUT DUE TO THE PRESENCE OF PRIMARY KEY, THE OUTPUT IS MEANINGLESS

SELECT * FROM 
(SELECT CUSTOMER_ID, PRODUCT,QUANTITY FROM PIVOT_TEST)
PIVOT XML (
SUM(QUANTITY) AS SUM_QUANTITY -- ALIAS 
FOR (PRODUCT) 
IN (SELECT DISTINCT PRODUCT FROM PIVOT_TEST));
-- WITH 'PIVOT XML' CLAUSE WE CAN USE DISTINCT CLAUSE TO FETCH THE VARIOUS GROUPS
-- TO BE DISPLAYED IN THE COLUMN, THIS IS NOT POSSIBLE WITH 'PIVOT' CLAUSE

SELECT * FROM 
(SELECT CUSTOMER_ID, PRODUCT,QUANTITY FROM PIVOT_TEST)
PIVOT XML (SUM(QUANTITY) AS SUM_QUANTITY 
FOR (PRODUCT) 
IN (ANY)); 
-- SIMILAR REASON AS ABOVE

SELECT * FROM UNPIVOT_TEST
UNPIVOT (
QUANTITY 
FOR PRODUCT_CODE -- COLUMN NAME OR ALIAS IN THE OUTPUT
IN (A_SUM_QUANTITY AS 'A', B_SUM_QUANTITY AS 'B', C_SUM_QUANTITY AS 'C'));

Use pattern matching to recognize patterns across multiple rows in a table

SELECT * FROM INVOICE 
MATCH_RECOGNIZE (
  PARTITION BY CUSTOMERID
  ORDER BY INVOICEDATE
  MEASURES  STRT.INVOICEDATE AS START_DATE,
            FINAL LAST(DOWN.INVOICEDATE) AS BOTTOM_DATE,
            FINAL LAST(UP.INVOICEDATE) AS END_DATE
  ONE ROW PER MATCH
  AFTER MATCH SKIP TO LAST UP
  PATTERN(STRT DOWN+ UP+)
  DEFINE 
     DOWN AS DOWN.TOTAL > PREV(DOWN.TOTAL),
     UP AS UP.TOTAL > PREV(UP.TOTAL)
)
  WHERE END_DATE - START_DATE > 12;
